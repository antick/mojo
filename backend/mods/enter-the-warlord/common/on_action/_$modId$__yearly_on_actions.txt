# $relates_to=common\on_action\yearly_on_actions.txt

# Called from code once a year for "playable" (count+) characters.
# Root is the character
yearly_playable_pulse = {
  on_actions = {
    _$modId$__create
    _$modId$__attack
  }
}

_$modId$__create = {
  trigger = {
    NOR = {
      has_trait = _$modId$__trait
      has_trait = craven
    }

    has_game_rule = _$modId$__player_yes
    is_independent_ruler = yes
    highest_held_title_tier >= tier_duchy
    is_adult = yes
    prestige_level >= 4
    piety_level >= 2

    OR = {
      has_trait = education_martial_1
      has_trait = education_martial_2
      has_trait = education_martial_3
      has_trait = education_martial_4
    }

    martial >= 18
  }

  effect = {
    random = {
      chance = 0
      modifier = {
        add = -3
        has_trait = education_martial_1
      }
      modifier = {
        add = 1
        has_trait = education_martial_4
      }
      modifier = {
        add = 5
        has_trait = ambitious
      }
      modifier = {
        add = -2
        is_ai = yes
      }
      modifier = {
        add = 1
        has_government = tribal_government
      }
      modifier = {
        add = 2
        has_government = clan_government
      }
      modifier = {
        add = -1000
        AND = {
          root = {
              is_ai = no
          }
          has_game_rule = _$modId$__player_no
        }
      }
      modifier = {
        add = 5
        has_game_rule = _$modId$__increased_little
      }
      modifier = {
        add = 10
        has_game_rule = _$modId$__increased_big
      }
      trigger_event = {
        id = _$modId$__event.0001
      }
    }
  }
}

_$modId$__attack = {
  trigger = {
    has_trait = _$modId$__trait
    is_ai = yes
    NOT = {
      is_at_war = yes
    }
  }

  effect = {
    trigger_event = {
      id = _$modId$__event.999
    }
  }
}
