
### Surveyor - Perks

mustering_the_troops_perk = {
	lifestyle = wanderer_lifestyle
	tree = surveyor
	position = { 2 0 }
	icon = node_wanderer
	
	auto_selection_weight = {
		value = 1
		if = {
			limit = {
				has_trait = lifestyle_traveler
			}
			add = 1000
		}
		if = {
			limit = {
				has_focus = wanderer_internal_affairs_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}

	name = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					NOT = { government_has_flag = government_is_settled }
				}
				desc = mustering_the_troops_perk_adventurer_name
			}
			desc = mustering_the_troops_perk_name
		}
	}

	effect = {
		if = {
			limit = {
				government_has_flag = government_is_settled
			}
			custom_description_no_bullet = {
				text = mustering_the_troops_perk_effect
			}
		}
		else = {
			custom_description_no_bullet = {
				text = mustering_the_troops_perk_2_effect
			}
		}
	}
}
